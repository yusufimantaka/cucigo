\section{Project Overview}
% ==========================================

\begin{frame}
    \frametitle{What is CuciGo?}
    \small
    \textbf{CuciGo} is a mobile-first laundry pickup and delivery marketplace connecting three user groups:

    \begin{columns}[T]
        \begin{column}{0.32\textwidth}
            \begin{tcolorbox}[colback=cucigoTeal!10,colframe=cucigoTeal,title=\centering\small Students]
                \scriptsize
                Browse services, place orders, track laundry, manage payments
            \end{tcolorbox}
        \end{column}

        \begin{column}{0.32\textwidth}
            \begin{tcolorbox}[colback=ugmBlue!10,colframe=ugmBlue,title=\centering\small Partners]
                \scriptsize
                Manage incoming orders, update status, set pricing
            \end{tcolorbox}
        \end{column}

        \begin{column}{0.32\textwidth}
            \begin{tcolorbox}[colback=cucigoCoral!10,colframe=cucigoCoral,title=\centering\small Drivers]
                \scriptsize
                Navigate routes, confirm pickups, update delivery status
            \end{tcolorbox}
        \end{column}
    \end{columns}

    \vspace{0.5em}
    \begin{tcolorbox}[colback=ugmLightGrey,colframe=ugmBlue,title=\small Key Facts]
        \scriptsize
        \begin{itemize}\setlength{\itemsep}{1pt}
            \item 30 functional requirements (FR-1 to FR-30)
            \item 4 user roles: Student, Partner, Driver, Admin
            \item Tech stack: Next.js, TypeScript, Tailwind CSS, Prisma + SQLite
        \end{itemize}
    \end{tcolorbox}
\end{frame}

\section{Lessons Learned}
% ==========================================

\begin{frame}
    \frametitle{Lessons Learned}
    \small
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \begin{tcolorbox}[colback=green!5,colframe=green!60!black,title=\small What Worked Well]
                \scriptsize
                \begin{itemize}\setlength{\itemsep}{1pt}
                    \item AI-assisted ideation rapidly produced a comprehensive SRS document following IEEE 830 standards
                    \item HTML prototypes enabled \textbf{visual stakeholder feedback} before writing any code
                    \item OpenCode's agentic harness allowed parallel file creation, dramatically reducing development time
                    \item Existing design tokens (CSS custom properties) translated cleanly to Tailwind CSS v4 theme variables
                    \item Prisma's type-safe client caught schema errors at compile time
                \end{itemize}
            \end{tcolorbox}
        \end{column}

        \begin{column}{0.48\textwidth}
            \begin{tcolorbox}[colback=red!5,colframe=red!70!black,title=\small Challenges]
                \scriptsize
                \begin{itemize}\setlength{\itemsep}{1pt}
                    \item Prisma v7 required driver adapters (breaking change from v6 patterns) --- required resolving import paths and API changes
                    \item AI-generated code needed \textbf{architectural review}: the agent sometimes chose suboptimal patterns that required manual correction
                    \item Tailwind CSS v4 uses a different configuration approach (\texttt{@theme inline}) than v3's \texttt{tailwind.config.js}
                    \item Mobile-first design requires careful responsive testing across viewports
                    \item Real-time features (GPS, push notifications) need actual backend services beyond the current prototype
                \end{itemize}
            \end{tcolorbox}
        \end{column}
    \end{columns}
\end{frame}
